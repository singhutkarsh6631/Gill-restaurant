🔧 Database Schema

-- Database: Utkarsh Singh Restaurant DB (IndexedDB)
-- Engine: Browser IndexedDB (persistent, no server needed)

TABLE reservations {
  id          INTEGER   AUTO_INCREMENT PRIMARY KEY
  name        TEXT      NOT NULL
  phone       TEXT      NOT NULL
  persons     INTEGER   DEFAULT 1
  date        DATE      NOT NULL
  time        TIME      NOT NULL
  message     TEXT      NULLABLE
  status      ENUM      ('pending', 'confirmed', 'cancelled') DEFAULT 'pending'
  created_at  DATETIME  DEFAULT CURRENT_TIMESTAMP
}

TABLE newsletter_subscribers {
  id            INTEGER   AUTO_INCREMENT PRIMARY KEY
  email         TEXT      UNIQUE NOT NULL
  subscribed_at DATETIME  DEFAULT CURRENT_TIMESTAMP
}

-- Indexes
CREATE INDEX idx_res_date     ON reservations(date);
CREATE INDEX idx_res_status   ON reservations(status);
CREATE UNIQUE INDEX idx_nl_email ON newsletter_subscribers(email);

Data is stored persistently in your browser's IndexedDB. To see it in DevTools: F12 → Application → IndexedDB → UtkarshRestaurantDB
